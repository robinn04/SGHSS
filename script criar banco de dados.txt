create database SGHSS;
use SGHSS;
CREATE TABLE Administrador (
  idAdministrador INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Login_idLogin INTEGER UNSIGNED NOT NULL,
  nomeAdministrador VARCHAR(100) NULL,
  cpfAdministrador CHAR(11) NULL,
  nivelAcesso INTEGER UNSIGNED NULL,
  PRIMARY KEY(idAdministrador),
  INDEX Administrador_FKIndex1(Login_idLogin)
);

CREATE TABLE Login (
  idLogin INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  usuario VARCHAR(50) NULL,
  senha VARCHAR(200) NULL,
  tipoUsuario VARCHAR(15) NULL,
  statusLogin BOOL NULL,
  PRIMARY KEY(idLogin)
);

CREATE TABLE ProfissionalSaude (
  idProfissionalSaude INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Login_idLogin INTEGER UNSIGNED NOT NULL,
  nomeProfissional VARCHAR(100) NULL,
  cpfProfissional CHAR(11) NULL,
  cargo VARCHAR(100) NULL,
  horarioTrabalho TIME NULL,
  PRIMARY KEY(idProfissionalSaude),
  INDEX ProfissionalSaude_FKIndex1(Login_idLogin)
);

CREATE TABLE Paciente (
  idPaciente INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Login_idLogin INTEGER UNSIGNED NOT NULL,
  Endereco_idEndereco INTEGER UNSIGNED NOT NULL,
  nomePaciente  VARCHAR(100) NULL,
  dataNacismentoPaciente DATE NULL,
  cpfPaciente CHAR(11) NULL,
  historicoClinico VARCHAR(250) NULL,
  PRIMARY KEY(idPaciente),
  INDEX Paciente_FKIndex1(Endereco_idEndereco),
  INDEX Paciente_FKIndex2(Login_idLogin)
);

CREATE TABLE Endereco (
  idEndereco INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  longradouro VARCHAR(100) NULL,
  numero CHAR(10) NULL,
  complemento VARCHAR(100) NULL,
  bairro VARCHAR(50) NULL,
  cidade VARCHAR(50) NULL,
  estado VARCHAR(50) NULL,
  PRIMARY KEY(idEndereco)
);

CREATE TABLE Consulta (
  idConsulta INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Paciente_idPaciente INTEGER UNSIGNED NOT NULL,
  ProfissionalSaude_idProfissionalSaude INTEGER UNSIGNED NOT NULL,
  dataHoraConsulta DATETIME NULL,
  tipoConsulta VARCHAR(200) NULL,
  statusConsulta BOOL NULL,
  observacoesConsultas VARCHAR(250) NULL,
  PRIMARY KEY(idConsulta),
  INDEX Consulta_FKIndex1(ProfissionalSaude_idProfissionalSaude),
  INDEX Consulta_FKIndex2(Paciente_idPaciente)
);

CREATE TABLE Telemedicina (
  idTelemedicina INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Paciente_idPaciente INTEGER UNSIGNED NOT NULL,
  ProfissionalSaude_idProfissionalSaude INTEGER UNSIGNED NOT NULL,
  linkVideo VARCHAR(250) NULL,
  dataHoraTelemedicina DATETIME NULL,
  duracaoTelemedicina TIME NULL,
  statusTelemedicina BOOL NULL,
  PRIMARY KEY(idTelemedicina),
  INDEX Consulta_FKIndex1(ProfissionalSaude_idProfissionalSaude),
  INDEX Consulta_FKIndex2(Paciente_idPaciente)
);

CREATE TABLE Estoque (
  idEstoque INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Administrador_idAdministrador INTEGER UNSIGNED NOT NULL,
  responsavel VARCHAR(100) NULL,
  ultimaAtualizacao DATETIME NULL,
  PRIMARY KEY(idEstoque),
  INDEX Estoque_FKIndex1(Administrador_idAdministrador)
);

CREATE TABLE Suprimentos (
  idSuprimentos INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Estoque_idEstoque INTEGER UNSIGNED NOT NULL,
  nomeSuprimento VARCHAR(50) NULL,
  tipoSuprimento VARCHAR(50) NULL,
  quantidade INTEGER UNSIGNED NULL,
  dataValidade DATE NULL,
  PRIMARY KEY(idSuprimentos),
  INDEX Suprimentos_FKIndex1(Estoque_idEstoque)
);

CREATE TABLE Exame (
  idExame INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Consulta_idConsulta INTEGER UNSIGNED NOT NULL,
  tipoExame VARCHAR(100) NULL,
  dataHoraExame DATETIME NULL,
  resultadoExame VARCHAR(200) NULL,
  statusExame BOOL NULL,
  PRIMARY KEY(idExame),
  INDEX Exame_FKIndex1(Consulta_idConsulta)
);

CREATE TABLE LogAuditoria (
  idLogAuditoria INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Administrador_idAdministrador INTEGER UNSIGNED NOT NULL,
  Login_idLogin INTEGER UNSIGNED NOT NULL,
  usuario VARCHAR(100) NULL,
  acaoLogAuditoria BOOL NULL,
  dataHoraLogAuditoria DATETIME NULL,
  ipOrigem VARCHAR(45) NULL,
  PRIMARY KEY(idLogAuditoria),
  INDEX LogAuditoria_FKIndex1(Login_idLogin),
  INDEX LogAuditoria_FKIndex2(Administrador_idAdministrador)
);

CREATE TABLE Backup (
  idBackup INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Administrador_idAdministrador INTEGER UNSIGNED NOT NULL,
  dataHoraExecucao DATETIME NULL,
  tipoBackup VARCHAR(50) NULL,
  statusBackup BOOL NULL,
  PRIMARY KEY(idBackup),
  INDEX Backup_FKIndex1(Administrador_idAdministrador)
);
